### 5.1 日志基础设施 — `app/logging_setup.py`（新增）

```python
"""统一日志配置。在 launcher.main() 最前调用 setup_logging()。"""
from __future__ import annotations
import logging
import logging.handlers
import os
from pathlib import Path

# Windows 标准路径；Mac 兜底（dev mode 用）
if os.name == "nt":
    LOG_DIR = Path(os.environ.get("LOCALAPPDATA", Path.home())) / "TA-F" / "PS-BATCH" / "logs"
else:
    LOG_DIR = Path.home() / ".tafpsd" / "logs"

LOG_FILE = LOG_DIR / "app.log"
MAX_BYTES = 2_000_000  # 2 MB
BACKUP_COUNT = 3       # rotating 2MB × 3 = 6MB 上限


def setup_logging(level: int = logging.INFO) -> Path:
    """配置 root logger，写到 rotating file。返回日志根目录用于 Help 菜单显示。"""
    try:
        LOG_DIR.mkdir(parents=True, exist_ok=True)
    except OSError:
        # 权限不够 → 回退到 TEMP，不要崩
        import tempfile
        fallback = Path(tempfile.gettempdir()) / "tafpsd_logs"
        fallback.mkdir(parents=True, exist_ok=True)
        globals()["LOG_DIR"] = fallback
        globals()["LOG_FILE"] = fallback / "app.log"

    fmt = "%(asctime)s [%(levelname)s] %(name)s: %(message)s"
    handler = logging.handlers.RotatingFileHandler(
        LOG_FILE, maxBytes=MAX_BYTES, backupCount=BACKUP_COUNT, encoding="utf-8",
    )
    handler.setFormatter(logging.Formatter(fmt))
    root = logging.getLogger()
    root.setLevel(level)
    root.addHandler(handler)
    root.info("=" * 60)
    root.info("PS-BATCH starting · log dir: %s", LOG_DIR)
    return LOG_DIR
```

### 5.4 诊断信息扩展 — `app/diagnostics.py`（新增）

把现有 `_copy_diagnostic_info`（launcher.py:1420）抽出来，扩抓 3 个日志源：

```python
"""统一诊断报告生成。一键拷贝粘贴给维护者用。"""
from __future__ import annotations
from pathlib import Path
import sys

from logging_setup import LOG_FILE
from persistence import PRESETS_PATH

DIAG_HEADER = """\
TA-F PSD DiffBatch diagnostic
=============================
Version:   {version}
Python:    {python}
PS:        {ps_app}
Presets:   {presets}
Log file:  {log_file}
{paths}
"""

def build_report(
    *, version: str, ps_app: str,
    after_path: str, before_path: str, folder_path: str, output_path: str,
    last_output_folder: str | None,
    gui_log_tail: list[str],
) -> str:
    """组合 3 个日志源 + 当前状态。返回纯文本。"""
    paths = (
        f"After:     {after_path or '(none)'}\n"
        f"Before:    {before_path or '(none)'}\n"
        f"Batch:     {folder_path or '(none)'}\n"
        f"Output:    {output_path or '(default)'}\n"
        f"Last out:  {last_output_folder or '(none)'}\n"
    )
    out = [DIAG_HEADER.format(
        version=version,
        python=sys.version.split()[0],
        ps_app=ps_app,
        presets=PRESETS_PATH,
        log_file=LOG_FILE,
        paths=paths,
    )]
    out.append("\n--- GUI log (last 50) ---\n")
    out.append("\n".join(gui_log_tail[-50:]) or "(empty)")
    out.append("\n\n--- app.log (last 200) ---\n")
    out.append(_tail(LOG_FILE, 200))
    if last_output_folder:
        normalize_log = Path(last_output_folder) / "normalize_log.txt"
        out.append(f"\n\n--- normalize_log.txt (last 100) [{normalize_log}] ---\n")
        out.append(_tail(normalize_log, 100))
        script_log = Path(last_output_folder).parent / "_script_log" / "script_run_log.txt"
        if script_log.exists():
            out.append(f"\n\n--- script_run_log.txt (last 100) ---\n")
            out.append(_tail(script_log, 100))
    return "\n".join(out)


def _tail(path: Path, n: int) -> str:
    try:
        if not path.exists():
            return "(not present)"
        lines = path.read_text(encoding="utf-8", errors="replace").splitlines()
        return "\n".join(lines[-n:])
    except Exception as exc:
        return f"(read failed: {exc})"
```

### 5.5 错误统一入口 — `app/errors.py`（新增）

```python
"""所有用户可见错误走这里。文案符合 §4.1 规范。"""
from __future__ import annotations
import logging
import traceback
from tkinter import messagebox

from logging_setup import LOG_FILE

log = logging.getLogger(__name__)


def show_error(parent, title: str, exc: Exception, *, retry=None) -> None:
    """短消息给用户 + 完整栈写日志。可选 retry callback。"""
    tb = traceback.format_exc()
    log.error("%s: %s\n%s", title, exc, tb)
    msg = str(exc) if str(exc) else exc.__class__.__name__
    full = (
        f"{msg}\n\n"
        f"详细日志：{LOG_FILE}\n"
        f"提交反馈时请用 Help → Copy diagnostic info。"
    )
    if retry:
        if messagebox.askretrycancel(title, full, parent=parent):
            retry()
    else:
        messagebox.showerror(title, full, parent=parent)
```

### 6.2 NAS manifest 真超时 — 修 `updater._read_manifest`（P1-2）

UNC IO 没有 socket timeout，必须后台线程 + 主线程死线：

```python
def _read_manifest(url_or_path: str, timeout: float) -> Optional[dict]:
    """3s 死线：UNC IO 阻塞也不让启动卡住。"""
    import threading
    box: dict = {"raw": None, "exc": None}
    done = threading.Event()

    def worker():
        try:
            if url_or_path.lower().startswith(("http://", "https://")):
                with urllib.request.urlopen(url_or_path, timeout=timeout) as resp:
                    box["raw"] = resp.read()
            else:
                box["raw"] = Path(url_or_path).read_bytes()
        except Exception as exc:
            box["exc"] = exc
        finally:
            done.set()

    threading.Thread(target=worker, daemon=True, name="manifest-probe").start()
    if not done.wait(timeout=timeout):
        log.info("manifest probe timed out after %ss", timeout)
        return None
    if box["exc"]:
        log.warning("manifest unreachable: %s", box["exc"])
        return None
    raw = box["raw"]
    if raw is None:
        return None
    try:
        return json.loads(raw.decode("utf-8"))
    except (UnicodeDecodeError, json.JSONDecodeError) as exc:
        log.warning("manifest malformed: %s", exc)
        return None
```

### 6.3 PS COM watchdog — 修 `ps_driver._drive_ps`（P1-1）

把同步 COM 调用放后台线程，主调用方等 timeout：

```python
def _drive_ps(jsx_path: Path, ps_app: str = "Photoshop.Application",
              timeout_secs: int = 1800) -> str:
    """COM 调用包 watchdog。超时抛 TimeoutError，文案提示用户手动关 PS。"""
    import threading
    try:
        import win32com.client  # type: ignore
        import pythoncom         # type: ignore
    except ImportError as exc:
        raise RuntimeError("pywin32 is required. Install: pip install pywin32") from exc

    box: dict = {"exc": None}
    done = threading.Event()

    def worker():
        try:
            pythoncom.CoInitialize()  # 子线程必须自己初始化 apartment
            ps_com = win32com.client.Dispatch("Photoshop.Application")
            js = '$.evalFile(new File("' + _to_jsx_string(str(jsx_path)) + '"));'
            ps_com.DoJavaScript(js)
        except Exception as exc:
            box["exc"] = exc
        finally:
            done.set()
            pythoncom.CoUninitialize()

    threading.Thread(target=worker, daemon=True, name="ps-com-call").start()
    if not done.wait(timeout=timeout_secs):
        raise TimeoutError(
            f"Photoshop 在 {timeout_secs}s 内未响应。请手动关闭 PS 后重试。"
            f"如重复发生请粘贴诊断信息（Help → Copy diagnostic info）给维护者。"
        )
    if box["exc"]:
        raise RuntimeError(f"Photoshop COM call failed: {box['exc']}") from box["exc"]
    return ""
```

**注意**：超时后 COM 调用仍在原 worker 线程里阻塞，daemon 线程会被进程退出时回收。UI 立即恢复。

"""产 exe 后启动 3 秒，检查进程未崩。捕 PyInstaller 打包链坏掉的情况。"""
import subprocess
import sys
import time
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
EXE = ROOT / "app" / "dist" / "TA-F PSD DiffBatch" / "TA-F PSD DiffBatch.exe"

if not EXE.exists():
    print(f"FAIL: exe not produced: {EXE}", file=sys.stderr)
    sys.exit(1)

proc = subprocess.Popen([str(EXE)])
try:
    time.sleep(3)
    if proc.poll() is not None:
        print(f"FAIL: exe exited immediately (rc={proc.returncode})", file=sys.stderr)
        sys.exit(1)
    print("OK: exe alive after 3s")
finally:
    proc.terminate()
    try:
        proc.wait(timeout=5)
    except subprocess.TimeoutExpired:
        proc.kill()

### 12.2 回滚策略

| 情况 | 操作 |
|---|---|
| 1.5.0 发布后发现 P0 bug | NAS releases\ 同位置回滚 1.4.1 setup.exe；改 latest.json version 回 1.4.1 + mandatory=true → 自动降级 |
| 用户升级后 preset 加载失败 | persistence 自动备份到 `.broken.<ts>.json` → TROUBLESHOOTING 教用户找 |
| 自动更新永远静默失败 | env `PSBATCH_UPDATE_URL` 临时改向（个别机器）；下一版本修 nas_config |

| 日志写满磁盘（极小概率，rotating 6MB 上限）| 用户清 `~/.tafpsd/logs/` 即可 |

### 12.3 升级路径

| 来源 | 目标 | 路径 |
|---|---|---|
| 1.4.1 | 1.5.0 | 自动更新弹窗 → Inno 静默安装；preset 文件兼容（schema 1→2 透明迁移）|
| 1.4.0 及更早（无自动更新） | 1.5.0 | 维护者手动通知；NAS 拷安装包；preset 兼容 |
| dev 模式 | 1.5.0 | `git pull` + `pip install -r requirements.lock.txt`，无安装包路径 |

### 12.4 备份策略

- NAS `releases\` 保留前 3 个版本 setup.exe（用于回滚）
- `~/.tafpsd_presets.json` 每次 schema 迁移前自动备份为 `.tafpsd_presets.json.bak.v<old_schema>`
- 损坏 preset 文件自动重命名为 `.tafpsd_presets.json.broken.<时间>.json`
- 日志自动 rotate（2MB×3，6MB 上限）

## 13 · 硬约束

- 不引入新运行时依赖（保持 customtkinter / tkinterdnd2 / psd-tools / Pillow / pywin32 不变）
- Windows-only（macOS dev 模式只为开发者方便，不出 .app）
- setup.exe 体积 ≤ 30MB（关 UPX 后预计 27MB）
- 首次启动可交互 ≤ 2s（生产 exe）
- NAS 不可达启动延迟 ≤ 3s（硬死线）
- PS COM 超时上限 1800s（D14 决策）
- 日志磁盘占用上限 6MB（rotating）
- 不破 1.4.1 的现有 preset 文件兼容（schema 迁移必须透明）
